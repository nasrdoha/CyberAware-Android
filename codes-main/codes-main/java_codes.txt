public class GameActivity extends AppCompatActivity {
    private TextView questionText;
    private TextView timerText;
    private Button option1;
    private Button option2;
    private Button option3;
    private Button option4;
    private ImageView questionImage;
    private final List<GameQuestion> allQuestions = new ArrayList();
    private final List<GameQuestion> filteredQuestions = new ArrayList();
    private final List<GameQuestion> selectedQuestions = new ArrayList();
    private int currentQuestionIndex = 0;
    private int correctCount = 0;
    private int wrongCount = 0;
    private String currentLevel = "BEGINNER";
    private CountDownTimer countDownTimer;
    private static final long QUESTION_TIME_MS = 25000L;

    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        this.setContentView(layout.activity_game);
        this.questionText = (TextView)this.findViewById(id.levelText);
        this.questionImage = (ImageView)this.findViewById(id.questionImage);
        this.timerText = (TextView)this.findViewById(id.timerText);
        this.option1 = (Button)this.findViewById(id.option1);
        this.option2 = (Button)this.findViewById(id.option2);
        this.option3 = (Button)this.findViewById(id.option3);
        this.option4 = (Button)this.findViewById(id.option4);
        this.currentLevel = this.getIntent().getStringExtra("level");
        if (this.currentLevel == null) {
            this.currentLevel = "BEGINNER";
        }

        this.loadQuestions();
        this.prepareQuestionsForCurrentLevel();
        this.option1.setOnClickListener((v) -> this.handleAnswerClick(0));
        this.option2.setOnClickListener((v) -> this.handleAnswerClick(1));
        this.option3.setOnClickListener((v) -> this.handleAnswerClick(2));
        this.option4.setOnClickListener((v) -> this.handleAnswerClick(3));
        this.showQuestion();
    }

    private String getGameQuestionsFile() {
        String lang = this.getSharedPreferences("Settings", 0).getString("lang", "en");
        return lang.equals("ar") ? "game_questions_ar.txt" : "game_questions.txt";
    }

    private void loadQuestions() {
        this.allQuestions.clear();

        try {
            File file = new File(this.getFilesDir(), this.getGameQuestionsFile());
            BufferedReader reader = new BufferedReader(new InputStreamReader(new FileInputStream(file), "UTF-8"));
            StringBuilder builder = new StringBuilder();

            String line;
            while((line = reader.readLine()) != null) {
                builder.append(line).append("\n");
            }

            reader.close();
            String data = builder.toString();
            String[] lines = data.split("\n");
            int j = 0;

            while(j < lines.length) {
                if (lines[j].trim().isEmpty()) {
                    ++j;
                } else if (!lines[j].trim().startsWith("LEVEL:")) {
                    ++j;
                } else {
                    String level = lines[j++].replace("LEVEL:", "").trim();
                    String type = lines[j++].replace("TYPE:", "").trim();
                    String question = lines[j++].trim();
                    String[] options = lines[j++].trim().split(",");
                    String answerLine = lines[j++].trim();
                    List<Integer> answers = new ArrayList();

                    for(String s : answerLine.split(",")) {
                        try {
                            answers.add(Integer.parseInt(s.trim()));
                        } catch (Exception var19) {
                        }
                    }

                    int imageResId = 0;
                    if (j < lines.length) {
                        String nextLine = lines[j].trim();
                        if (!nextLine.isEmpty() && !nextLine.startsWith("LEVEL:") && !nextLine.startsWith("TYPE:") && !nextLine.toLowerCase().startsWith("note")) {
                            if (!nextLine.equals("0")) {
                                imageResId = this.getResources().getIdentifier(nextLine, "drawable", this.getPackageName());
                            }

                            ++j;
                        }
                    }

                    String explanation = "";
                    if (j < lines.length) {
                        String noteLine = lines[j].trim().toLowerCase();
                        if (noteLine.startsWith("note")) {
                            explanation = lines[j].replace("note:", "").replace("note :", "").trim();
                            ++j;
                        }
                    }

                    if ("MCQ".equalsIgnoreCase(type)) {
                        this.allQuestions.add(new GameQuestion(question, options, answers, imageResId, level, type, explanation));
                    }
                }
            }
        } catch (Exception e) {
            e.printStackTrace();
        }

    }

    private void prepareQuestionsForCurrentLevel() {
        this.filteredQuestions.clear();
        this.selectedQuestions.clear();

        for(GameQuestion q : this.allQuestions) {
            if (q.getLevel() != null && q.getLevel().trim().equalsIgnoreCase(this.currentLevel.trim()) && "MCQ".equalsIgnoreCase(q.getType())) {
                this.filteredQuestions.add(q);
            }
        }

        Collections.shuffle(this.filteredQuestions);
        int count = Math.min(20, this.filteredQuestions.size());

        for(int i = 0; i < count; ++i) {
            this.selectedQuestions.add((GameQuestion)this.filteredQuestions.get(i));
        }

    }

    private void showQuestion() {
        if (this.selectedQuestions.isEmpty()) {
            this.showNoQuestionsDialog();
        } else if (this.currentQuestionIndex >= this.selectedQuestions.size()) {
            this.finishGame();
        } else {
            GameQuestion q = (GameQuestion)this.selectedQuestions.get(this.currentQuestionIndex);
            this.questionText.setText(q.getQuestion());
            String[] options = q.getOptions();
            this.option1.setText(options.length > 0 ? options[0] : "");
            this.option2.setText(options.length > 1 ? options[1] : "");
            this.option3.setText(options.length > 2 ? options[2] : "");
            this.option4.setText(options.length > 3 ? options[3] : "");
            if (q.getImageResId() != 0) {
                this.questionImage.setVisibility(0);
                this.questionImage.setImageResource(q.getImageResId());
            } else {
                this.questionImage.setVisibility(8);
            }

            this.option1.setVisibility(0);
            this.option2.setVisibility(0);
            this.option3.setVisibility(0);
            this.option4.setVisibility(0);
            this.startTimer();
        }
    }

    private void startTimer() {
        if (this.countDownTimer != null) {
            this.countDownTimer.cancel();
        }

        if (this.timerText != null) {
            this.timerText.setText("20");
        }

        this.countDownTimer = (new CountDownTimer(25000L, 1000L) {
            public void onTick(long millisUntilFinished) {
                if (GameActivity.this.timerText != null) {
                    GameActivity.this.timerText.setText(String.valueOf(millisUntilFinished / 1000L));
                }

            }

            public void onFinish() {
                if (GameActivity.this.timerText != null) {
                    GameActivity.this.timerText.setText("0");
                }

                ++GameActivity.this.wrongCount;
                ++GameActivity.this.currentQuestionIndex;
                GameActivity.this.showQuestion();
            }
        }).start();
    }

    private void handleAnswerClick(int selectedIndex) {
        if (!this.selectedQuestions.isEmpty() && this.currentQuestionIndex < this.selectedQuestions.size()) {
            if (this.countDownTimer != null) {
                this.countDownTimer.cancel();
            }

            GameQuestion q = (GameQuestion)this.selectedQuestions.get(this.currentQuestionIndex);
            int correctAnswer = this.getCorrectAnswerIndex(q);
            if (selectedIndex == correctAnswer) {
                ++this.correctCount;
                this.showFeedbackDialog(this.getString(string.correct), true, q, correctAnswer);
            } else {
                ++this.wrongCount;
                this.showFeedbackDialog(this.getWrongAnswerMessage(q, correctAnswer), false, q, correctAnswer);
            }

        }
    }

    private int getCorrectAnswerIndex(GameQuestion q) {
        return q.getCorrectAnswers() != null && !q.getCorrectAnswers().isEmpty() ? (Integer)q.getCorrectAnswers().get(0) : 0;
    }

    private String getWrongAnswerMessage(GameQuestion q, int correctAnswerIndex) {
        StringBuilder message = new StringBuilder();
        message.append(this.getString(string.wrong)).append("\n\n");
        message.append("✔ ").append(this.getString(string.correct_answer)).append(" ");
        message.append(q.getOptions()[correctAnswerIndex]);
        if (q.getExplanation() != null && !q.getExplanation().trim().isEmpty()) {
            message.append("\n\n\ud83d\udca1 ").append(this.getString(string.note)).append("\n");
            message.append(q.getExplanation());
        }

        return message.toString();
    }

    private void showFeedbackDialog(String message, boolean correct, GameQuestion q, int correctAnswerIndex) {
        TextView msg = new TextView(this);
        msg.setText(message);
        msg.setTextSize(16.0F);
        msg.setTextColor(-16777216);
        msg.setPadding(40, 30, 40, 30);
        msg.setTextDirection(4);
        msg.setGravity(8388611);
        (new AlertDialog.Builder(this, 16974394)).setTitle(correct ? this.getString(string.excellent) : this.getString(string.incorrect)).setView(msg).setCancelable(false).setPositiveButton(this.getString(string.next), (d, i) -> {
            ++this.currentQuestionIndex;
            this.showQuestion();
        }).show();
    }

    private void finishGame() {
        if (this.countDownTimer != null) {
            this.countDownTimer.cancel();
        }

        int total = this.selectedQuestions.size();
        int percentage = total == 0 ? 0 : (int)((double)this.correctCount * (double)100.0F / (double)total);
        if (percentage >= 75) {
            String finalLevel = this.getString(string.level_advanced);
            String feedback = this.getString(string.feedback_advanced);
        } else if (percentage >= 50) {
            String finalLevel = this.getString(string.level_intermediate);
            String feedback = this.getString(string.feedback_intermediate);
        } else {
            String finalLevel = this.getString(string.level_beginner);
            String feedback = this.getString(string.feedback_beginner);
        }

        Intent intent = new Intent(this, ResultActivity.class);
        intent.putExtra("assessmentLevel", this.currentLevel);
        intent.putExtra("percentage", percentage);
        this.startActivity(intent);
        this.finish();
    }

    private void showNoQuestionsDialog() {
        if (this.countDownTimer != null) {
            this.countDownTimer.cancel();
        }

        (new AlertDialog.Builder(this)).setTitle(this.getString(string.game_finished)).setMessage(this.getString(string.no_questions_available)).setCancelable(false).setPositiveButton(this.getString(string.ok), (d, i) -> this.finish()).show();
    }

    protected void onDestroy() {
        if (this.countDownTimer != null) {
            this.countDownTimer.cancel();
        }

        super.onDestroy();
    }
}

package com.example.mygame;

public class MainActivity extends AppCompatActivity {
    Button startButton;

    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        this.setContentView(layout.activity_main);
        this.copyRawToInternal(raw.questions, "questions.txt");
        this.copyRawToInternal(raw.questions_ar, "questions_ar.txt");
        this.copyRawToInternal(raw.game_questions, "game_questions.txt");
        this.copyRawToInternal(raw.game_questions_ar, "game_questions_ar.txt");
        this.startButton = (Button)this.findViewById(id.startButton);
        this.startButton.setOnClickListener((v) -> {
            Intent intent = new Intent(this, SecondActivity.class);
            this.startActivity(intent);
        });
    }

    private void copyRawToInternal(int rawId, String fileName) {
        try {
            File file = new File(this.getFilesDir(), fileName);
            if (file.exists()) {
                return;
            }

            InputStream inputStream = this.getResources().openRawResource(rawId);
            FileOutputStream outputStream = new FileOutputStream(file);
            byte[] buffer = new byte[1024];

            int length;
            while((length = inputStream.read(buffer)) > 0) {
                outputStream.write(buffer, 0, length);
            }

            inputStream.close();
            outputStream.close();
        } catch (Exception e) {
            e.printStackTrace();
        }

    }
}
